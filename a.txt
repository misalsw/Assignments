Move files from one folder to respective Folders.

$ nano copyfiles.sh

Swapnil Misal@Swaps MINGW64 ~/Desktop/Bridgelabz/Assignment (master)
$ cat copyfiles.sh
for file in `ls *.txt`;
do
        folderName=`echo $file | awk -F. '{print $1}'`;
        mkdir $folderName;
        mv $file $folderName;
        echo Copied $file to $folderName/;
done
